给定两个单链表，编写算法找出两个链表的公共节点

（1）给节点增加一个区域，如果访问过这个节点，做上标志，哪个节点被访问两次，即为公共阶段。时间复杂度O(m+n)

（2）假设两个单链表长度一致，公共阶段后的长度相同，所以同步比较节点、找到公共节点即可。

当两个单链表长度不一致时，先遍历长的链表直到两个链表长度一致时，开始同步遍历。

```c
/**
*给定两个单链表，编写算法找出两个链表的公共节点
*/

typedef struct Node {
    int data;
    struct Node *next;
}Node, *LinkList;

//构造单链表(无头节点)(头插法)
LinkList creatLinkList(LinkList L) {
    int x;
    L = (LinkList)malloc(sizeof(Node));
    Node *s;
    scanf("%d", &x);
    L->data = x;
    L->next = NULL;    //构造第一个节点
    scanf("%d", &x);
    while(x!=9999) {
        s = (Node*)malloc(sizeof(Node));
        s->data = x;
        s->next = L;
        L = s;
        scanf("%d", &x);
    }
    return L;
}

//构造两个有公共节点的单链表，用于测试
void testLinkList(LinkList L1, LinkList L2) {
    L2->next->next = L1->next;
}

int lengthLinkList(LinkList L) {
    int length = 0;
    while(L) {
        length++;
        L = L->next;
    }
    return length;
}

//寻找两个单链表的公共节点
Node* searchCommonNode(LinkList L1, LinkList L2) {
    int length1 = lengthLinkList(L1);
    int length2 = lengthLinkList(L2);
    printf("%d %d", length1, length2);
    int dist;
    if(length1 > length2) {
        dist = length1 - length2;
        while(dist-- > 0) {
            L1 = L1->next;
        }
    }
    else if(length2 > length1) {
        dist = length2 - length1;
        while(dist-- > 0) {
            L2 = L2->next;
        }
    }
    while(L1!=NULL && L1!=L2) {
        L1 = L1->next;
        L2 = L2->next;
    }
    return L1;
}

void printfLinkList(LinkList L) {
    while (L!=NULL) {
    printf("%d ", L->data);
    L = L->next;
    }
}

int main(int argc, const char * argv[]) {
    LinkList L1;
    LinkList L2;
    L1 = creatLinkList(L1);
    printf("\n输出list1：");
    printfLinkList(L1);
    L2 = creatLinkList(L2);
    printf("\n输出list2：");
    printfLinkList(L2);
    testLinkList(L1, L2);
    printf("\n输出list2：");
    printfLinkList(L2);
    L1 = searchCommonNode(L1, L2);
    printf("\n公共节点是：%p, 值为：%d\n", L1, L1->data);
    return 0;
}
```

涉及算法：

1. 倒序输出单链表：一种是将元素遍历，存放在数组中，按照倒序输出；另一种是利用递归的方式输出

2. 删除最小值节点（假设唯一）

3. 就地逆置，辅助空间为O(1)：一种是暴力解决，两个for循环嵌套；另一种是利用头插法依次将后续节点插入头结点之后，达到逆置的目的。

4. 无序带头结点单链表，删除s,t之间的值

5. 将带头结点无序单链表，变为递增有序：一种是暴力解决，两个for循环嵌套；另一种是利用直接插入排序法依次将后续节点插入合适的位置

```c

#include <stdio.h>
#include <stdlib.h>

typedef struct Node{
	int data;
	struct Node *next;
}Node, *LinkList;

//头插法构建单链表 
void creatLinkList(LinkList L) {
	Node* s;
	int x;
	scanf("%d", &x);
	while(x!=9999) {
		s = (Node*)malloc(sizeof(Node));
		s->data = x;
		s->next = L->next;
		L->next = s;
		L->data ++;
		scanf("%d", &x);
	}
	printf("length:%d", L->data);
}

//顺序输出单链表 
void printLinkList(LinkList L) {
	while(L->next) {
		printf("%d ", L->next->data);
		L = L->next;
	}
}

//倒序输出单链表
void reversePrintLinkList(LinkList L) {
	int data[L->data];
	int i = 0;
	while(L->next) {
		data[i++] = L->next->data;
		L = L->next;
	}
	while(i > 0) {
		printf("%d ", data[--i]);
	}
}

//倒序输出全部元素（递归的方法）
void reversePrintLinkList_2(LinkList L) {
    if(L->next) {
        reversePrintLinkList_2(L->next);
    }
    printf("%d ",L->data);
    
}

//删除最小值节点（假设唯一） 
void deletMin(LinkList L) {
	Node *p = L; 
	Node* s = (Node*)malloc(sizeof(Node));
	if(L->next == NULL) return;
	s = L;
	while(L->next) {
		if(L->next->data < s->next->data) {
			s = L;
		} 
		L = L->next;
	}
	printf("\n被删除的数：%d",s->next->data);
	//删除最小值点的过程
	Node *t = s->next; 
	s->next = t->next;
	free(t);
	p->data--;
} 

void exchange(Node* p) {
	int temp;
	temp = p->data;
	p->data = p->next->data;
	p->next->data = temp;
}

//就地逆置，辅助空间为O(1)，时间复杂度O(n^2) 
void reverseLinkList(LinkList L) {	
	int i, j;
	Node* s = L; 
	int length = L->data;
	L = L->next;
	for(i = 1; i < length; i++) {
		L = s->next;		
		for(j = 0; j < length-i ; j++) {
			exchange(L);
			L = L->next;
		}
	}
}

//带头节点的单链表就地逆置（参照头插法形成单链表）
void reverseLinkList_2(LinkList L) {
    if(L->next == NULL)    return;
    Node *p = L->next, *r;
    L->next = NULL;
    while(p) {
        r = p->next;
        p->next = L->next;
        L->next = p;
        p = r;
    }
}

//无序带头结点单链表，删除s,t之间的值
void deletBetwST(LinkList L, int s, int t) {
	if(s>=t)	return;
	Node* q = L;
	Node* P = (Node*)malloc(sizeof(Node));
	while(L->next) {
		if(L->next->data > s && L->next->data < t) {    //满足条件，删除节点 
			P = L->next;
			L->next = L->next->next;
			free(P);
			q->data--;
		} else
			L = L->next;
	}	
} 

//将带头结点无序单链表，变为递增有序 
void sortLinkList(LinkList L) {
	int i, j;
	Node* s = L; 
	int length = L->data;
	L = L->next;
	for(i = 1; i < length; i++) {
		L = s->next;		
		for(j = 1; j < length ; j++) {
			if(L->data > L->next->data) 
				exchange(L);
			L = L->next;
		}
	}
}

//利用直接插入排序，将带头节点单链表变为递增有序
void sortLinkList_2(LinkList L) {
    Node *p = L->next->next, *r, *s;
    L->next->next = NULL;
    while(p!=NULL) {
        r = p->next;
        s = L;
        while(s->next!=NULL && s->next->data < p->data) {
            s = s->next;
        }
        p->next = s->next;    //先将被插入节点的后继指针确定，不需要中间参数
        s->next = p;
        p = r;
    }

}

int main(int argc, char *argv[]) {
	LinkList L;
	L = (LinkList)malloc(sizeof(Node));
	L->next = NULL;
	L->data = 0;
	creatLinkList(L);
	printf("\n原始链表：");
	printLinkList(L);
	printf("\n倒序输出链表1：");
	reversePrintLinkList(L);
	printf("\n倒序输出链表2：");
	reversePrintLinkList_2(L->next);
	deletMin(L);
	printf("\n删除最小值后的链表：");
	printLinkList(L);
	reverseLinkList(L);
	printf("\n就地逆置链表1：");
	printLinkList(L);
	reverseLinkList_2(L);
	printf("\n就地逆置链表2：");
	printLinkList(L);
	deletBetwST(L, 4,20);
	printf("\n删除4和20之间的数的链表：");
	printLinkList(L);
	sortLinkList(L); 
	printf("\n递增有序排列后的链表：");
	printLinkList(L);
	return 0;
}
```
