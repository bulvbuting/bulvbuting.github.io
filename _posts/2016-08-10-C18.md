题目描述：  

循环队列入队，出队:

方法1：牺牲一个单元

方法2：增加一个标志域

> 方法1

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

/*
以循环队列作为demo，因为顺序队列容易产生“假溢出”，循环队列则可以避免这种情况 
入队时，少入一个单元来区分队空和队满 
*/ 
#define Maxsize 5
using namespace std;
typedef struct {
	int data[Maxsize];
	int rear, front;
}SqQueue;
SqQueue Q;

//循环队列初始化 
void InitQueue() {
	Q.rear = Q.front = 0;
}

//判空，如果是空，返回true 
bool IsEmpty() {
	if(Q.rear == Q.front)    return true;
	return false;
} 

//入队 
bool EnQueue(int x) {
	if((Q.rear+1)%Maxsize == Q.front)	return false;    //队满，出栈失败
	Q.data[Q.rear] = x;
	Q.rear = (Q.rear+1)%Maxsize;
	return true; 
}

//出队 
bool DeQueue(int &x) {
	if(Q.rear == Q.front) {	   //队空，出栈失败
		x = -1;
	    return false;    
	}
	x = Q.data[Q.front];
	Q.front = (Q.front+1)%Maxsize;
	return true; 
} 

int main(int argc, char** argv) {
	InitQueue();
	bool flag;
	int x = 0;
	flag = IsEmpty();
	cout<<"判队空："<<flag<<endl;
	flag = EnQueue(1);
	cout<<"1 入队状态："<<flag<<endl;
	flag = EnQueue(2);
	cout<<"2 入队状态："<<flag<<endl;
	flag = EnQueue(3);
	cout<<"3 入队状态："<<flag<<endl;
	flag = EnQueue(4);
	cout<<"4 入队状态："<<flag<<endl;
	flag = EnQueue(5);
	cout<<"5 入队状态："<<flag<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	return 0;
}
```

> 方法2

```c

#include <iostream>
#include <stdlib.h>
#include <stdio.h>
#define MaxSize 5
using namespace std;

typedef struct {
	int data[MaxSize];
	int rear, front;
	int tag;
}SqQueue;

SqQueue Q;

//初始化 
void InitSqQueue() {
	Q.rear = Q.front = 0;
	Q.tag = 0;
}

//入队 
bool EnQueue(int x) {
	if(Q.rear == Q.front && Q.tag == 1)     return false;
	Q.data[Q.rear] = x;
	Q.rear = (Q.rear+1)%MaxSize;
	Q.tag = 1;
	return true;
}

//出队 
bool DeQueue(int &x) {
	x = -1;
	if(Q.rear == Q.front && Q.tag == 0)    return false;
	x = Q.data[Q.front];
	Q.front = (Q.front+1)%MaxSize;
	Q.tag = 0;
	return true;
} 

int main(int argc, char** argv) {
	InitSqQueue();
	bool flag;
	int x = 0;
	flag = EnQueue(1);
	cout<<"1 入队状态："<<flag<<endl;
	flag = EnQueue(2);
	cout<<"2 入队状态："<<flag<<endl;
	flag = EnQueue(3);
	cout<<"3 入队状态："<<flag<<endl;
	flag = EnQueue(4);
	cout<<"4 入队状态："<<flag<<endl;
	flag = EnQueue(5);
	cout<<"5 入队状态："<<flag<<endl;
	flag = EnQueue(6);
	cout<<"6 入队状态："<<flag<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	return 0;
}
```
