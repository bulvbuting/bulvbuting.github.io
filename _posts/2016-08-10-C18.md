使用工具：DEV C++

题目描述：  

循环队列入队，出对

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

/*
以循环队列作为demo，因为顺序队列容易产生“假溢出”，循环队列则可以避免这种情况 
入队时，少入一个单元来区分对空和对满 
*/ 
#define Maxsize 5
using namespace std;
typedef struct {
	int data[Maxsize];
	int rear, front;
}SqQueue;
SqQueue Q;

//循环对列初始化 
void InitQueue() {
	Q.rear = Q.front = 0;
}

//判空，如果是空，返回true 
bool IsEmpty() {
	if(Q.rear == Q.front)    return true;
	return false;
} 

//入队 
bool EnQueue(int x) {
	if((Q.rear+1)%Maxsize == Q.front)	return false;    //对满，出栈失败
	Q.data[Q.rear] = x;
	Q.rear = (Q.rear+1)%Maxsize;
	return true; 
}

//出队 
bool DeQueue(int &x) {
	if(Q.rear == Q.front) {	   //对空，出栈失败
		x = -1;
	    return false;    
	}
	x = Q.data[Q.front];
	Q.front = (Q.front+1)%Maxsize;
	return true; 
} 

int main(int argc, char** argv) {
	InitQueue();
	bool flag;
	int x = 0;
	flag = IsEmpty();
	cout<<"判队空："<<flag<<endl;
	flag = EnQueue(1);
	cout<<"1 入队状态："<<flag<<endl;
	flag = EnQueue(2);
	cout<<"2 入队状态："<<flag<<endl;
	flag = EnQueue(3);
	cout<<"3 入队状态："<<flag<<endl;
	flag = EnQueue(4);
	cout<<"4 入队状态："<<flag<<endl;
	flag = EnQueue(5);
	cout<<"入队状态："<<flag<<endl;
	flag = DeQueue(x);
	cout<<"出对状态："<<flag<<"；出对值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出对状态："<<flag<<"；出对值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出对状态："<<flag<<"；出对值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出对状态："<<flag<<"；出对值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出对状态："<<flag<<"；出对值："<<x<<endl;
	return 0;
}

```
