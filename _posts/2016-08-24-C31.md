题目描述：

根据中序、先序遍历结果构造二叉链表；利用层次遍历获取二叉树的深度。

利用层次遍历获取二叉树的深度。

先序遍历结果：{1, 2, 4, 3, 5, 6, 7, 8};

中序遍历结果：{4, 2, 1, 5, 3, 8, 7, 6};

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#define MaxSize 10

using namespace std;

typedef struct BiTNode {
	int data;
	struct BiTNode *lchild, *rchild;
}BiTNode, *BiTree;

typedef struct Queue {
	struct BiTNode *data[MaxSize];
	int front, rear;
}Queue; 

BiTree T;
Queue Q;

void visit(BiTNode *node) {
    cout<<(node->data)<<" ";
}

//队初始化
void InitQueue() {
	Q.rear = Q.front = -1;
} 

//入队 
void enQueue(BiTNode *node) {
	if((Q.rear+1)%MaxSize == Q.front)    return;
	Q.rear = (Q.rear+1)%MaxSize;
	Q.data[Q.rear] = node;
}

//出队
void deQueue(BiTNode *&node) {
	if(Q.front == Q.rear)    return;
	Q.front = (Q.front+1)%MaxSize;
	node = Q.data[Q.front];	
}

bool IsEmptyQueue() {
	if(Q.front == Q.rear)    return true;
	return false;
}

//先序访问递归
void PreOrder(BiTree T) {
    if (T!=NULL) {
        visit(T);
        PreOrder(T->lchild);
        PreOrder(T->rchild);
    }
}

//中序访问递归
void InOrder(BiTree T) {
    if (T!=NULL) {
        InOrder(T->lchild);
        visit(T);
        InOrder(T->rchild);
    }
}

BiTree ToBiTree(int *A, int *B, int i1, int h1, int i2, int h2) {
	int i;
	BiTNode *T = (BiTNode*)malloc(sizeof(BiTNode));
	T->data = *(A+i1);
	for(i=i2; *(B+i)!=*(A+i1); i++);    //根结点的位置，这一句是关键。 
	int lengthl = i-i2;
	int lengthr = h2-i;
	if(lengthl>0)
		T->lchild = ToBiTree(A, B, i1+1, i1+lengthl, i2, i2+lengthl-1);
	else    T->lchild = NULL;
	if(lengthr>0)
		T->rchild = ToBiTree(A, B, i1+lengthl+1, i1+lengthl+lengthr, i+1, i+lengthr);
	else    T->rchild = NULL;
	return T;
}

//利用层次遍历获取二叉树的深度（根结点深度为1）
int getTreeLevel(BiTree T) {
	int level = 0;
	int last = 0;    //每行最后一个元素的下标 
	InitQueue();
	BiTNode *p = T;
	if(p!=NULL) {
		enQueue(p);
		while(!IsEmptyQueue()) {
			deQueue(p);
			if(p->lchild!=NULL)
				enQueue(p->lchild);
			if(p->rchild!=NULL)
				enQueue(p->rchild);
			if(Q.front == last) {
				level++;
				last = Q.rear;
			}
		}
	}
	return level;
}

int main(int argc, char** argv) {
	int A[8] = {1, 2, 4, 3, 5, 6, 7, 8};
	int B[8] = {4, 2, 1, 5, 3, 8, 7, 6};
	int length = sizeof(A)/sizeof(int); 
	BiTree T = ToBiTree(A, B, 0, 7, 0, 7);
	cout<<"先序递归访问二叉树：";
    PreOrder(T);
    cout<<endl<<"中序递归访问二叉树：";
    InOrder(T);
    cout<<endl<<"层次遍历得到二叉树的高度：";
    int level = getTreeLevel(T);
    cout<<level<<endl;
	return 0;
}
```
