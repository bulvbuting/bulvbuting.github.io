题目描述：

一个递增有序单链表，去掉重复元素 （逻辑和顺序表一样）

```c

#include <stdio.h>
#include <stdlib.h>

/**
* 一个递增有序单链表，去掉重复元素 （逻辑和顺序表一样） 
*/

typedef struct Node {
	int data;
	struct Node *next;
}Node, *LinkList; 

/**
* 头插法构建单链表，链表顺序与输入顺序相反，因为单链表为递增的，所以输入时顺序应该为递减的 
*/ 
void creatLinkList(LinkList L) {
    L->next = NULL;
    int x;
    Node *s;
    scanf("%d", &x);
    while(x!=9999) {
        s = (Node*)malloc(sizeof(Node));
        s->data = x;
        s->next = L->next;
        L->next = s;
        scanf("%d", &x);
    }
}

void printLinkList(LinkList L) {
    if(L->next == NULL)    return;
    while(L->next!=NULL) {
        printf("%d ", L->next->data);
        L = L->next;
    }
}

//去重主逻辑 
void duplicateLinkList(LinkList L) {
	if(L->next == NULL)    return;
	Node *p = L->next->next;
	L->next->next = NULL;    //将L初始化为只有一个结点
	L = L->next;
	while(p!=NULL) {
		if(p->data == L->data)
			p = p->next;
		else {
			L->next = p;
			p = p->next;
			L = L->next;	
		}	
	} 
	L->next = NULL;
}

int main(int argc, char *argv[]) {
	LinkList L = (LinkList)malloc(sizeof(Node));
	creatLinkList(L);
	printf("输出创建的链表（递减输入）：");
	printLinkList(L);
	duplicateLinkList(L);
	printf("\n去重后的链表：");
	printLinkList(L);
	return 0;
}
```
