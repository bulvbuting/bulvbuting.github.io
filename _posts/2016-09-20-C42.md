### 改进排序中的划分算法，使每次选取的枢轴值都是随机地从当前字表中选取的

算法代码：

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

void coutArray(int *a, int lo, int hi) {
	for(int i = lo; i <= hi; i++)
		cout<<a[i]<<" ";
	cout<<endl;
}

//交换a,b的值 
void exchange(int &a, int &b) {
	if(a==b)    return;
	a = a^b;
	b = a^b;
	a = a^b;
}

//双向冒泡排序（第一趟把最大的元素放在最后面，第二次把最小的元素放在最前面，交替进行） 
void sort(int *a, int lo, int hi) {
	int flag = 1;
	while(lo < hi) {
		int i = lo, j = hi;
		while(i < hi) {
			if(a[i] > a[i+1]) {
				exchange(a[i], a[i+1]);
				flag = 0;
			}
			i++;
		} 
		hi--;
		
		if(flag == 1)    return;
		
		while(j > lo) {
			if(a[j] < a[j-1]) {
				exchange(a[j], a[j-1]);
				flag = 0;
			}
			j--;
		}
		lo++;
		
		if(flag == 1)    return;
	}
}

//随机获得枢轴值
int getPart(int *a, int lo, int hi) {
	if(lo>=hi)    return -1;
	int random = lo + rand() % (hi-lo+1);    //之前int rand = lo + rand() % (hi-lo+1);报错的原因是rand这个变量不能重复定义
	if(random!=lo)	exchange(a[lo], a[random]);
	int v = a[lo];
	int i = lo, j = hi;
	while(i < j) {
		while(i<j && a[j] >= v) j--;
		a[i] = a[j];
		while(i<j && a[i] <= v) i++;
		a[j] = a[i];
	}
	a[i] = v;
	return i;
}

//获取L[1……n]中第k小（从小到大排序的第k个元素） 
int getMinK(int *a, int lo, int hi, int k) {
	if(k > hi-lo+1 || k < 1)    return -1;
	int p = getPart(a, lo, hi);
	while(p!=k-1) {
		if(p > k-1)
			p = getPart(a, lo, p-1);
		else 
			p = getPart(a, p+1, hi);			
	}
	return a[p];
}

int main(int argc, char** argv) {
	int a[10] = {6,2,3,4,1,9,8,5,10,7};
	int k = 8;
	int minK = getMinK(a, 0, 9, k);
	cout<<"第"<<k<<"小的元素是："<<minK<<endl; 
	sort(a, 0, 9);
	coutArray(a, 0, 9);
	return 0;
}
```

### 将所有的奇数都移到偶数之前(类似快速排序的分割算法)

算法代码：

```c
#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

void coutArray(int *a, int lo, int hi) {
	for(int i = lo; i <= hi; i++)
		cout<<a[i]<<" ";
	cout<<endl;
}

//将所有的奇数都移到偶数之前
void sortNumbers(int *a, int lo, int hi) {
	if(hi<=lo) return;
	int v = a[lo];
	while(lo < hi) {
		while(lo < hi && !(a[hi]%2)) hi--;
		a[lo] = a[hi];
		while(lo < hi && a[lo]%2) lo++;
		a[hi] = a[lo];
	}
	a[lo] = v;
} 

int main(int argc, char** argv) {
	int a[10] = {6,2,3,4,1,9,8,5,10,7};
	sortNumbers(a, 0, 9);
	coutArray(a, 0, 9);
	return 0;
}
```

### 荷兰国旗问题：一个仅由红白蓝三种颜色的条块组成的条块序列，涉及一个O(n)复杂度的算法，使这些条块按红白蓝的顺序排好

![](/assets/img/2016-09-18-C41/1.png)

算法代码：

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

void coutArray(int *a, int lo, int hi) {
	for(int i = lo; i <= hi; i++)
		cout<<a[i]<<" ";
	cout<<endl;
}

//交换a,b的值 
void exchange(int &a, int &b) {
	if(a==b)    return;
	a = a^b;
	b = a^b;
	a = a^b;
}

//荷兰国旗问题（类似三向切分的分割方法）
void sortThree(int *a, int lo, int hi) {
	if(hi <= lo)	return;
	int p = lo, i = lo, j = hi;
	int v = a[lo];    //以lo为基准
	while(p <= j) {
		if(a[p] < v) 
			exchange(a[p++], a[i++]);
		else if(a[p] > v) 
			exchange(a[p], a[j--]);
		else 
			p++;
	}
} 

int main(int argc, char** argv) {
	int a[10] = {6,2,3,4,1,9,8,5,10,7};
	sortThree(a, 0, 9);
	coutArray(a, 0, 9);
	return 0;
}

```
