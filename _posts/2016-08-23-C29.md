使用工具：DEV C++

题目描述：

分别用递归和非递归写出二叉树的前序遍历、中序遍历、后序遍历。其中后序遍历最为复杂。

构造二叉树时，输入为-1表示空树。

```c

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

typedef struct BiTNode {
    int data;
    int flag;    //该节点是否访问过的标记，针对后序非递归访问访问。0：没访问，1：访问了 
    struct BiTNode *lchild, *rchild;
}BiTNode, *BiTree;

typedef struct stack {
    struct BiTNode* data[10];
    int top;
}stack;

BiTree T;
stack S;

void InitStack() {
    S.top = 0;
}

void push(stack &S, BiTNode *node) {
    if(node == NULL)    return;
    S.data[S.top++] = node;
}

void pop(stack &S, BiTNode *&node) {
    node = S.data[--S.top];
}

void getTop(stack &S, BiTNode *&node) {
	node = S.data[S.top-1];
}

//生成二叉树 
void createBiTree(BiTree &T) {
	int a;
	scanf("%d", &a);    //头结点 
	if(a!=-1) {
		T = (BiTNode*)malloc(sizeof(BiTNode));
	    T->data = a;
	    T->lchild = T->rchild = NULL;
	    createBiTree(T->lchild);
	    createBiTree(T->rchild);
	} else {
		T = NULL;
	}
}

void visit(BiTNode *node) {
    cout<<(node->data)<<" ";
}

//先序访问递归
void PreOrder(BiTree T) {
    if (T!=NULL) {
        visit(T);
        PreOrder(T->lchild);
        PreOrder(T->rchild);
    }
}

//先序访问非递归
void PreOrder2(BiTree T) {
    InitStack();
    BiTree p = T;    //遍历指针
    while(p!=NULL||S.top!=0) {
        if(p!=NULL) {
            visit(p);
            push(S, p->rchild);
            p = p->lchild;
        } else {
            pop(S, p);
            visit(p);
            push(S, p->rchild);
            p = p->lchild;
        }
    }
}

//中序访问递归
void InOrder(BiTree T) {
    if (T!=NULL) {
        InOrder(T->lchild);
        visit(T);
        InOrder(T->rchild);
    }
}

//中序访问非递归
void InOrder2(BiTree T) {
    InitStack();
    BiTree p = T;    //遍历指针
    while(p!=NULL||S.top!=0) {
        if (p!=NULL) {
            push(S, p);
            p = p->lchild;
        } else {
            pop(S, p);
            visit(p);
            p = p->rchild;
        }
    }
}

//后序访问递归
void LastOrder(BiTree T) {
    if (T!=NULL) {
        LastOrder(T->lchild);
        LastOrder(T->rchild);
        visit(T);
    }
}

//后序访问非递归（复杂）
void LastOrder2(BiTree T) {
    InitStack();
    BiTree p = T;    //遍历指针根
    while (p!=NULL||S.top!=0) {
		if(p!=NULL) {
			push(S, p);
			if(p->lchild!=NULL)
				p = p->lchild;
			else
				p = p->rchild;
		} else {
			getTop(S, p);
			if(p->rchild == NULL || p->rchild->flag == 1) {    //没有右孩子或者右孩子已经访问了，访问该结点。 
			//如果有左孩子，入栈晚于其父辈节点 ，访问早于父辈结点，肯定已经访问过了。
				pop(S, p);
				visit(p);
				p->flag = 1;
				p = NULL;
			} else {    //有右孩子，并且没有被访问，则整个右子树都没有被访问。
				p = p->rchild;				
			}
		}
    }
}

int main(int argc, const char * argv[]) {
    createBiTree(T);
    cout<<"先序递归访问二叉树：";
    PreOrder(T);
    cout<<endl<<"先序非递归访问二叉树：";
    PreOrder2(T);
    cout<<endl<<"中序递归访问二叉树：";
    InOrder(T);
    cout<<endl<<"中序非递归访问二叉树：";
    InOrder2(T);
    cout<<endl<<"后序递归访问二叉树：";
    LastOrder(T);
    cout<<endl<<"后序非递归访问二叉树：";
    LastOrder2(T);
    return 0;
}
```
