1.方法1

```c

#include <stdio.h>
void main() {
	int i,j,k;
	for(i = 0; i < 16; i++) {	//共16行 ,上半个三角形 
		for(j = 0; j < 15-i; j++) {
			printf(" ");
		}
		for(k = 0; k < 2*i+1; k++) {
			printf("*");
		}
		printf("\n"); 
	}
	
	for(i = 0; i < 15; i++) {	//共15行，下半个三角形  
		for(j = 0; j < i+1; j++) {
			printf(" ");
		}
		for(k = 0; k < 29-2*i; k++) {
			printf("*");
		}
		printf("\n");
	} 
}

```

2.方法2

```c

#include <stdio.h>
void main() {
	int i,j,k,m,n;
	for(i = 0; i < 31; i++) {
		if(i<=15) {
			m = 15-i;
			n = 2*i+1;
		} else {
			m = i-15;
			n = 61-2*i;
		}
		for(j = 0; j < m; j++) {
			printf(" ");
		}	
		for(k = 0; k < n; k++) {
			printf("*");
		}
		printf("\n");
	}
}

```
