题目描述：

1. m,n为指向二叉树的两个结点的指针，求这两个结点最近公共祖先的指针

2. 求非空二叉树的宽度。（即具有结点数最多对那一行）

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <vector>
#define MaxSize 20

using namespace std;

typedef struct BiTNode {
    int data;
    struct BiTNode *lchild, *rchild;
    int flag;
}BiTNode, *	BiTree;

typedef struct Stack {
    struct BiTNode *data[MaxSize];
    int top;
}Stack;

typedef struct Queue {
    struct BiTNode *data[MaxSize];
    int front, rear;
}Queue;

BiTree T;
Stack S;
Queue Q;

void InitStack() {
    S.top = 0;
}

BiTNode* getTop() {
    return S.data[S.top-1];
}

void pushStack(BiTNode *p) {
    if(S.top == MaxSize)    return;
    S.data[S.top++] = p;
}

void popStack(BiTNode *&p) {
    if(S.top == 0)    return;
    p = S.data[--S.top];
}

void InitQueue() {
    Q.front = Q.rear = 0;
}

bool IsEmptyQueue() {
    if(Q.front == Q.rear)    return true;
    return false;
}

bool IsFullQueue() {
    if((Q.rear+1)%MaxSize == Q.front)    return true;
    return false;
}

void enQueue(BiTNode *p) {
    if(p == NULL)    return;
    if(!IsFullQueue()) {
        Q.data[Q.rear] = p;
        Q.rear = (Q.rear+1)%MaxSize;
    }
}

void deQueue(BiTNode *&p) {
    if(!IsEmptyQueue()) {
        p = Q.data[Q.front];
        Q.front = (Q.front+1)%MaxSize;
    }
}

void createTree(BiTree &T) {
    int data;
    scanf("%d", &data);
    if(data!=-1) {
        T = (BiTree)malloc(sizeof(BiTNode));
        T->data = data;
        createTree(T->lchild);
        createTree(T->rchild);
    } else    T = NULL;
}

//查找m指针对应结点的所有祖先(后序遍历)
vector<BiTree> printParents(BiTree T, BiTNode *m) {
    vector<BiTree> a;
    InitStack();
    BiTNode *p = T;
    while(p!=NULL || S.top!=0) {
        while(p!=NULL && p!=m) {
        	p->flag = 0;
            pushStack(p);
            p = p->lchild;
        }
        
        if(p == m) {    //找到指针m对应的结点
        	a.push_back(p);    //将原结点入向量 
            for(int i = 0; S.top!=0; i++) {
                popStack(p);
                a.push_back(p);
            }
            p = NULL;
        }
        
        while(S.top!=0 && p==NULL){    //栈不为空
            p = getTop();
            if(p->rchild== NULL) {    //注意区分结点左、右结点的访问情况
                popStack(p);
                p = getTop();
                if(p->flag == 1) {
                    popStack(p);
                    p = getTop();
                }
            }
            p->flag = 1;
            p = p->rchild;
        }
    }
    return a;
}


//查找m，n的最近的公共祖先结点
void findParent(BiTree T, BiTNode *m, BiTNode *n, BiTNode *&r) {
    r = (BiTree)malloc(sizeof(BiTNode));
    vector<BiTree> a = printParents(T, m);
    vector<BiTree> b = printParents(T, n);
    vector<BiTree>::iterator it, it1;
    for(int i=0; i<a.size(); i++) {
        for(int j=0; j<b.size(); j++) {
            if(a[i] == b[j]) {
                r = a[i];
                return;
            }
        }
    }
}

void visit(BiTNode *p) {
    cout<<p->data<<" ";
}

//先序遍历
void preTree(BiTree T) {
    if(T!=NULL) {
        visit(T);
        preTree(T->lchild);
        preTree(T->rchild);
    }
}

//中序遍历
void inTree(BiTree T) {
    if(T!=NULL) {
        inTree(T->lchild);
        visit(T);
        inTree(T->rchild);
    }
}

//获取非空二叉树的宽度
int getWidth(BiTree T) {
    if(T == NULL)
        return 0;
    int width = 1;
    BiTNode *p = T;
    enQueue(p);
    int pp = Q.rear;
    while(!IsEmptyQueue()) {
        while (Q.front!=pp) {
            deQueue(p);
            enQueue(p->lchild);
            enQueue(p->rchild);
        }
        width = (Q.rear+MaxSize-Q.front)%MaxSize>width?(Q.rear+MaxSize-Q.front)%MaxSize:width;
        pp = Q.rear;
    }
    return width;
}

int main(int argc, char** argv) {
    createTree(T);
    cout<<"先序遍历结果："; 
    preTree(T);
    cout<<endl<<"中序遍历结果：";
    inTree(T);
    cout<<endl<<"二叉树的宽度：";
    int width = getWidth(T);
    cout<<width;
    BiTNode *m = T->rchild->rchild, *n = T->rchild->rchild->rchild, *r;
    findParent(T, m, n, r);
    cout<<endl<<"m:"<<m->data<<"和n:"<<n->data<<"的最近公共祖先的地址："<<r<<endl<<"对应的结点值："<<r->data;
    return 0;
}
```
