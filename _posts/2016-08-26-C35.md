题目描述：

1. m,n为指向二叉树的两个结点的指针，求这两个结点最近公共祖先的指针

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <vector> 
#define MaxSize 10

using namespace std;

typedef struct BiTNode {
	int data;
	struct BiTNode *lchild, *rchild; 
}BiTNode, *	BiTree;

typedef struct Stack {
	struct BiTNode *data[MaxSize];
	int top;
}Stack;

BiTree T;
Stack S;

void InitStack() {
	S.top = 0;
}

void pushStack(BiTNode *p) {
	if(S.top == MaxSize)    return;
	S.data[S.top++] = p;
}

void popStack(BiTNode *&p) {
	if(S.top == 0)    return;
	p = S.data[--S.top];
}

void createTree(BiTree &T) {
	int data;
	scanf("%d", &data);
	if(data!=-1) {
		T = (BiTree)malloc(sizeof(BiTNode));
		T->data = data;
		createTree(T->lchild);
		createTree(T->rchild);
	} else    T = NULL;
}

//查找m指针对应结点的所有祖先(后序遍历)
vector<BiTree> printParents(BiTree T, BiTNode *m) {
	vector<BiTree> a;
	if(T == NULL)    return a;
	InitStack();
	BiTNode *p = T;

		if(p!=NULL) {
			if(p == m) {    //找到指针m对应的结点

					a.push_back(p);
				}

}

//先序遍历 
void preTree(BiTree T) {
	if(T!=NULL) {
		visit(T);
		preTree(T->lchild);
		preTree(T->rchild);
	} 
}

//中序遍历

	if(T!=NULL) {
		inTree(T->lchild);
