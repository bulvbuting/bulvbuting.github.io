```c
#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

typedef struct LNode {
    int data;
    LNode *next;
    LNode(int data):data(data),next(NULL){}
}LNode, *LinkList;

void createLinkList(LinkList &L) {
    LinkList S = L;
    int a;
    scanf("%d", &a);
    LNode *p;
    while(a!=-1) {
        p = new LNode(a);
        S->next = p;
        S = p;
        scanf("%d", &a);
    }
}

void coutLinkList(LinkList L) {
    while (L->next!=NULL) {
        cout<<L->next->data<<" ";
        L = L->next;
    }
    cout<<endl;
}

//将链表中<v的数放在链表前面部分，>=v的数放在链表后面，并保持相对顺序不变
void partiton(LinkList &L, int v) {
    LinkList p1 = (LinkList)malloc(sizeof(LNode));
    LinkList p2 = (LinkList)malloc(sizeof(LNode));
    p1->next = p2->next = NULL;
    LinkList temp1 = p1, temp2 = p2;
    while(L->next!=NULL) {
        if (L->next->data < v) {
            p1->next = L->next;
            p1 = p1->next;
        } else {
            p2->next = L->next;
            p2 = p2->next;
        }
        L = L->next;
    }
    p1->next = temp2->next;
    p2->next = NULL;
    L = temp1;
}

int main(int argc, const char * argv[]) {
    LinkList L = (LinkList)malloc(sizeof(LNode));
    createLinkList(L);
    partiton(L, 5);
    coutLinkList(L);
    return 0;
}
```

算法代码：

```c

//
//  main.cpp
//  18
//
//  Created by 赵琳 on 16/10/9.
//  Copyright © 2016年 赵琳. All rights reserved.
//

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

typedef struct LNode {
    int data;
    LNode *next;
    LNode(int data):data(data),next(NULL){}
}LNode, *LinkList;

void createLinkList(LinkList &L) {
    LinkList S = L;
    int a;
    scanf("%d", &a);
    LNode *p;
    while(a!=-1) {
        p = new LNode(a);
        S->next = p;
        S = p;
        scanf("%d", &a);
    }
}

void createLinkList2(LinkList &L1, LinkList L2) {
    LinkList S = L1;
    int a;
    scanf("%d", &a);
    LNode *p;
    while(a!=-1) {
        p = new LNode(a);
        S->next = p;
        S = p;
        scanf("%d", &a);
    }
    S->next = L2->next;
}

void coutLinkList(LinkList L) {
    while (L->next!=NULL) {
        cout<<L->next->data<<" ";
        L = L->next;
    }
    cout<<endl;
}

int getLength(LinkList L) {
    int a = 0;
    while(L->next!=NULL) {
        a++;
        L = L->next;
    }
    return a;
}

//给定两个单链表，输出两个单链表的第一个公共结点。
void findFirstSameLNode(LinkList L1, LinkList L2) {
    int m = getLength(L1);
    int n = getLength(L2);
    if(m==0||n==0)
        cout<<"任意链表长度不能为0！"<<endl;
    int a;
    LinkList p1, p2;
    if(m>n) {
        p1 = L1;
        p2 = L2;
        a = m-n;
    } else {
        p1 = L2;
        p2 = L1;
        a = n-m;
    }   //p1对应比较长的链表,p2对应比较短的链表
    
    while(a!=0) {
        p1 = p1->next;
        a--;
    }
    
    while(p1!=NULL) {
        if(p1!=p2) {
            p1 = p1->next;
            p2 = p2->next;
        } else {
            cout<<"公共结点是："<<p1->data<<endl;
            break;
        }
    }
}

int main(int argc, const char * argv[]) {
    LinkList L = (LinkList)malloc(sizeof(LNode));
    cout<<"输入链表公共部分："<<endl;
    createLinkList(L);  //  公共部分，坠到两个单链表后面
    LinkList L1 = (LinkList)malloc(sizeof(LNode));
    LinkList L2 = (LinkList)malloc(sizeof(LNode));
    cout<<"输入链表非公共部分："<<endl;
    createLinkList2(L1, L);
    createLinkList2(L2, L);
    cout<<"最后生成的链表："<<endl;
    coutLinkList(L1);
    coutLinkList(L2);
    findFirstSameLNode(L1, L2);
    return 0;
}
```
