#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

typedef struct LNode {
    int data;
    LNode *next;
    LNode(int data):data(data),next(NULL){}
}LNode, *LinkList;

void createLinkList(LinkList &L) {
    LinkList S = L;
    int a;
    scanf("%d", &a);
    LNode *p;
    while(a!=-1) {
        p = new LNode(a);
        S->next = p;
        S = p;
        scanf("%d", &a);
    }
}

void coutLinkList(LinkList L) {
    while (L->next!=NULL) {
        cout<<L->next->data<<" ";
        L = L->next;
    }
    cout<<endl;
}

//将链表中<v的数放在链表前面部分，>=v的数放在链表后面，并保持相对顺序不变
void partiton(LinkList &L, int v) {
    LinkList p1 = (LinkList)malloc(sizeof(LNode));
    LinkList p2 = (LinkList)malloc(sizeof(LNode));
    p1->next = p2->next = NULL;
    LinkList temp1 = p1, temp2 = p2;
    while(L->next!=NULL) {
        if (L->next->data < v) {
            p1->next = L->next;
            p1 = p1->next;
        } else {
            p2->next = L->next;
            p2 = p2->next;
        }
        L = L->next;
    }
    p1->next = temp2->next;
    p2->next = NULL;
    L = temp1;
}

int main(int argc, const char * argv[]) {
    LinkList L = (LinkList)malloc(sizeof(LNode));
    createLinkList(L);
    partiton(L, 5);
    coutLinkList(L);
    return 0;
}

#include <stdlib.h>

using namespace std;

typedef struct LNode {
    int data;

}LNode, *LinkList;

    LinkList S = L;

    LNode *p;
    while(a!=-1) {
        p = new LNode(a);
        S->next = p;
        S = p;
        scanf("%d", &a);
    }
}

void createLinkList2(LinkList &L1, LinkList L2) {
    LinkList S = L1;
    int a;
    scanf("%d", &a);
    LNode *p;
    while(a!=-1) {
        p = new LNode(a);
        S->next = p;
        S = p;
        scanf("%d", &a);
    }
    S->next = L2->next;
}

void coutLinkList(LinkList L) {
    while (L->next!=NULL) {
        cout<<L->next->data<<" ";
        L = L->next;
    }
    cout<<endl;
}

int getLength(LinkList L) {
    int a = 0;
    while(L->next!=NULL) {
        a++;
        L = L->next;
    }
    return a;
}

//给定两个单链表，输出两个单链表的第一个公共结点。
void findFirstSameLNode(LinkList L1, LinkList L2) {
    int m = getLength(L1);
    int n = getLength(L2);
    if(m==0||n==0)
        cout<<"任意链表长度不能为0！"<<endl;
    int a;
    LinkList p1, p2;
    if(m>n) {
        p1 = L1;
        p2 = L2;
        a = m-n;
    } else {
        p1 = L2;
        p2 = L1;
        a = n-m;
    }   //p1对应比较长的链表,p2对应比较短的链表
    
    while(a!=0) {
        p1 = p1->next;
        a--;
    }
    
    while(p1!=NULL) {
        if(p1!=p2) {
            p1 = p1->next;
            p2 = p2->next;
        } else {
            cout<<"公共结点是："<<p1->data<<endl;
            break;
        }
    }
}

int main(int argc, const char * argv[]) {
    LinkList L = (LinkList)malloc(sizeof(LNode));
    cout<<"输入链表公共部分："<<endl;
    createLinkList(L);  //  公共部分，坠到两个单链表后面
    LinkList L1 = (LinkList)malloc(sizeof(LNode));
    LinkList L2 = (LinkList)malloc(sizeof(LNode));
    cout<<"输入链表非公共部分："<<endl;
    createLinkList2(L1, L);
    createLinkList2(L2, L);
    cout<<"最后生成的链表："<<endl;
    coutLinkList(L1);
    coutLinkList(L2);
    findFirstSameLNode(L1, L2);
    return 0;
}

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//最长括号匹配

typedef struct Stack {
    int data[10];
    int top;
}Stack;

void initStack(Stack &s) {
    s.top = 0;
}

bool isEmpty(Stack s) {
    return s.top==0;
}

void push(Stack &s, int v) {
    s.data[s.top++] = v;
}

int pop(Stack &s) {
    return s.data[--s.top];
}

int getTop(Stack s) {
    return s.data[s.top-1];
}

int getMaxLength(string a) {
    Stack s;
    initStack(s);
    int length = (int)a.length();
    int maxsize = 0,temp = -1;    //pp终止位
    for(int i = 0; i < length; i++) {
        if(a[i]=='(')
            push(s,i);
        else {
            if(isEmpty(s))
                temp = i;
            else {
                pop(s);
                if(isEmpty(s))
                    maxsize = maxsize>=i-temp?maxsize:i-temp;
                else
                    maxsize = maxsize>=i-getTop(s)?maxsize:i-getTop(s);
            }
        }
    }
    return maxsize;
}

int main(int argc, const char * argv[]) {
    string a1 = "(()";
    int length = getMaxLength(a1);
    cout<<a1<<"最大匹配长度为："<<length<<endl;
    
    string a2 = "()()";
    length = getMaxLength(a2);
    cout<<a2<<"最大匹配长度为："<<length<<endl;
    
    string a3 = "()(())";
    length = getMaxLength(a3);
    cout<<a3<<"最大匹配长度为："<<length<<endl;
    
    string a4 = "(()())";
    length = getMaxLength(a4);
    cout<<a4<<"最大匹配长度为："<<length<<endl;
    return 0;
}

//
//  main.cpp
//  18
//
//  Created by 赵琳 on 16/10/9.
//  Copyright © 2016年 赵琳. All rights reserved.
//

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

#define MaxSize 100

using namespace std;

typedef struct Stack {
    int data[MaxSize];
    int top;
}Stack;

void initStack(Stack &S) {
    S.top = 0;
}

void pushStack(Stack &S, int a) {
    S.data[S.top++] = a;
}

int popStack(Stack &S) {
    int data = S.data[--S.top];
    return data;
}

//检查a是不是符号，是的话返回1，否则返回0
bool check(const char *a) {
    if(a == "+" || a == "-" || a == "*" || a == "/")
        return 1;
    return 0;
}

int getResult(int p1, const int p2, const char *a) {
    int result = 0;
    if(a == "+")
        result = p1+p2;
    else if(a == "-")
        result = p1-p2;
    else if(a == "*")
        result = p1*p2;
    else if(a == "/")
        result = p1/p2;
    return result;
}

int getCharLength(const char *a[]) {
    int length = 0;
    while(a[length]!=NULL)
        length++;
    return length;
}

int Math(Stack S, const char *a[]) {
    initStack(S);
    int n = getCharLength(a);
    int result = 0;
    int p1, p2;
    for(int i = 0; i<n; i++) {
        if(!check(a[i])) {   //不是符号，入栈
            pushStack(S, atoi(a[i]));
        } else {  //是符号，出栈两个值
            p2 = popStack(S);
            p1 = popStack(S);
            result = getResult(p1, p2, a[i]);
            pushStack(S, result);
        }
    }
    
    if(S.top!=1)    //最后栈中应该只有一个结果值
        cout<<"输入纪录有误！"<<endl;
    return result;
}

int main(int argc, const char * argv[]) {
    Stack S;
    const char* a[5] = {"2","3","+","8","*"};
    int result= Math(S, a);
    cout<<"的计算结果为："<<result<<endl;
    return 0;
}
