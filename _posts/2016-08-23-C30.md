使用工具：DEV C++

题目描述：

借助栈和队列实现二叉树的由下到上，由右到左的层次遍历。

顺序层次遍历少了最后入栈的步骤，出队后，直接访问。

求二叉树的深度时，仿照中序遍历的非递归方法写的，添加一个深度域，表示结点所处位置的深度。

构造二叉树时，输入为-1表示空树。

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#define MaxSize 10

using namespace std;

typedef struct BiTNode {
	int data;
	int deep;    //为了非递归求二叉树深度添加 
	struct BiTNode *lchild, *rchild;
}BiTNode, *BiTree;

typedef struct stack {
	struct BiTNode* data[MaxSize];
    int top;
}stack;

typedef struct QueueNode {    //循环队列，借助队列 
	struct BiTNode *data[MaxSize];
	int front, rear;
}Queue;
 
BiTree T;
stack S;
QueueNode Q;

void InitStack() {
    S.top = 0;
}

void InitQueue() {
    Q.front = Q.rear = 0;
}

bool IsEmptyStack() {
	if(S.top == 0)    return true;
	return false;
}

bool IsEmptyQueue() {
	if(Q.rear == Q.front)    return true;
	return false;
}

void push(stack &S, BiTNode *node) {
    if(node == NULL)    return;
    S.data[S.top++] = node;
}

void pop(stack &S, BiTNode *&node) {
	if(S.top == 0)    return;
    node = S.data[--S.top];
}

void enQueue(BiTNode *&node) {
	if((Q.rear+1)%MaxSize == Q.front)    return;
	Q.data[Q.rear] = node;
	Q.rear = (Q.rear+1)%MaxSize;
}

void deQueue(BiTNode *&node) {
	if(Q.rear == Q.front)    return;
	node = Q.data[Q.front];
	Q.front = (Q.front+1)%MaxSize;
} 

//生成二叉树 
void createBiTree(BiTree &T) {
	int a;
	scanf("%d", &a);    //头结点 
	if(a!=-1) {
	    T = (BiTNode*)malloc(sizeof(BiTNode));
	    T->data = a;
	    T->lchild = T->rchild = NULL;
	    createBiTree(T->lchild);
	    createBiTree(T->rchild);
	} else {
		T = NULL;
	}
}

void visit(BiTNode *node) {
    cout<<(node->data)<<" ";
}

//先序访问递归
void PreOrder(BiTree T) {
    if (T!=NULL) {
        visit(T);
        PreOrder(T->lchild);
        PreOrder(T->rchild);
    }
}

//中序访问递归
void InOrder(BiTree T) {
    if (T!=NULL) {
        InOrder(T->lchild);
        visit(T);
        InOrder(T->rchild);
    }
}

//自下而上，自右而左的层次遍历 
void reverOrder() {
	BiTNode *p = T;
	if(T!=NULL) {
		enQueue(p);
	}
	
	while(!IsEmptyQueue()) {
		deQueue(p);
		if(p->lchild!=NULL)    enQueue(p->lchild);
		if(p->rchild!=NULL)    enQueue(p->rchild);
		push(S, p);
	}
	
	while(!IsEmptyStack()) {
		pop(S, p);
		visit(p);
	}
}

//递归求树的深度 
int deepBiTree(BiTree T) {
	int deep = 0; 
	if(T!=NULL) {
		int deepl = deepBiTree(T->lchild);
		int deepr = deepBiTree(T->rchild);
		deep = deepl>deepr?deepl+1:deepr+1;    //这层树的深度肯定比其子树深度大1 
	} 
	return deep;
}

//非递归求树的深度 
int deepBiTree2(BiTree T) {
	BiTNode *p = T;
	InitStack();
	int deep = 0;
	int deepMax = deep;
	while(p!=NULL||S.top!=0) {
		if(p!=NULL) {
			p->deep = ++deep;
			push(S, p);
			p = p->lchild;
		} else {
			pop(S, p);
			deep = p->deep; 
			p = p->rchild;
		}
		deepMax = deepMax>deep?deepMax:deep;
	}
	return deepMax;
}

int main(int argc, char** argv) {
	createBiTree(T);
	InitStack();
	InitQueue();
	cout<<"先序递归访问二叉树：";
	PreOrder(T);
	cout<<endl<<"中序递归访问二叉树：";
	InOrder(T);
	cout<<endl<<"自下而上，自右而左的层次遍历：";
	reverOrder();
	int a = deepBiTree(T);
	cout<<endl<<"递归遍历二叉树的深度："<<a; 
	a = deepBiTree2(T);
	cout<<endl<<"非递归遍历二叉树的深度："<<a<<endl; 
	return 0;
}
```
