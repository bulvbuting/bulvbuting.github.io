使用工具：xcode

题目描述：

两个栈共享一个存储区［0，……，maxsize－1］，设计有关入栈和出栈的方法


```c

//
//  main.cpp
//  10
//
//  Created by 赵琳 on 16/8/9.
//  Copyright © 2016年 赵琳. All rights reserved.
//

#include <iostream>
#include <stdio.h>
#include <stdlib.h>


#define maxsize 5
using namespace std;

typedef struct Stack {
    int a[maxsize];
    int top[2];
}Stack;
Stack s;

//i指插入共享栈的哪部分，分为 0,1两种
void push(int x, int i) {
    if(i<0||i>1) {
        cout<<"进栈方向参数输入错误！"<<endl;
        return;
    }

    if(s.top[1]-s.top[0] == 1) {
        cout<<"栈满，进栈失败！"<<endl;
        return;
    }

    if(i==0) {//左侧栈进栈
        s.a[++s.top[i]] = x;
    } else if(i==1) {//右侧栈进栈
        s.a[--s.top[i]] = x;
    }
}

//将出栈的值用x返回来
void pop(int &x, int i) {
    if(i<0||i>1) {
        cout<<"出栈方向参数输入错误！"<<endl;
        x = -1;
        return;
    }

    if(i==0) {
        if(s.top[i] == -1) {
            cout<<"栈空，出栈失败！"<<endl;
            x = -1;
            return;
        }
        x = s.a[s.top[i]--];
    } else if(i==1) {
    if(s.top[i] == maxsize) {
        cout<<"栈空，出栈失败！"<<endl;
        x = -1;
        return;
    }
        x = s.a[s.top[i]++];
    }
}

    int main(int argc, char** argv) {
    s.top[0] = -1;
    s.top[1] = maxsize;
    push(1, 0);
    push(1, 8);    //进栈方向参数输入错误！
    push(2, 0);
    push(3, 0);
    push(4, 1);
    push(5, 1);
    push(6, 1);    //栈满，进栈失败！
    int x = 0;
    pop(x, 1);
    cout<<x<<endl;
    pop(x, 1);
    cout<<x<<endl;
    pop(x, 1);    //1号栈只有两个元素，栈空出栈失败，返回－1
    cout<<x<<endl;
    pop(x, 0);
    cout<<x<<endl;
    pop(x, 0);
    cout<<x<<endl;
    pop(x, 0);
    cout<<x<<endl;
    pop(x, 0);    //0号栈只有三个元素，栈空出栈失败，返回－1
    cout<<x<<endl;
    return 0;
}



```
