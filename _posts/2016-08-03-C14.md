使用工具：DEV C++

题目描述：

将带头结点的单链表A ，拆分成两个单链表A, B. 其中序号为奇数的存在A中， 序号为偶数的存在B中 

```c

#include <stdio.h>
#include <stdlib.h>

/**
* 将带头结点的单链表A ，拆分成两个单链表A, B. 其中序号为奇数的存在A中， 序号为偶数的存在B中 
*/

typedef struct Node {
	int data;
	struct Node *next;
}Node, *LinkList; 

void creatLinkList(LinkList L) {
    L->next = NULL;
    int x;
    Node *s;
    scanf("%d", &x);
    while(x!=9999) {
        s = (Node*)malloc(sizeof(Node));
        s->data = x;
        s->next = L->next;
        L->next = s;
        scanf("%d", &x);
    }
}

void printLinkList(LinkList L) {
    if(L->next == NULL)    return;
    while(L->next!=NULL) {
        printf("%d ", L->next->data);
        L = L->next;
    }
}

void splitLinkList(LinkList A, LinkList B) {
	Node* L = A ;
	int i = 0;
	while(L->next!=NULL) {
		i++;
		if(i%2 == 1) {    //i为奇数 ，存在A 中 
			A->next = L->next;
			A = A->next;
		} else {    //i为奇数 ，存在A 中 
			B->next = L->next;
			B = B->next;
		} 
		L = L->next;
	}
	A->next = NULL;
	B->next = NULL;
} 

int main(int argc, char *argv[]) {
	LinkList A = (LinkList)malloc(sizeof(Node));
	creatLinkList(A);
	printf("输出创建的链表：");
	printLinkList(A); 
	LinkList B = (LinkList)malloc(sizeof(Node));
	splitLinkList(A, B);
	printf("\n输出拆分后的链表：\n");
	printLinkList(A); 
	printf("\n"); 
	printLinkList(B); 
	return 0;
}

```
