折半查找a是否在有序数组中,如果在，返回位置，否则返回-1(注意是有序数组)

算法代码：

```c

//
//  main.cpp
//  44_

//  Copyright © 2017年 赵琳. All rights reserved.
//

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//折半查找a是否在有序数组中,如果在，返回位置，否则返回-1(注意是有序数组)
int find(const int* arry, int size, int a) {
    bool isFind = false;
    int from = 0, to = size-1;
    int nindex = 0;
    while(from<=to) {
        nindex = (from+to)/2;
        if(a == arry[nindex]) {
            isFind = true;
            break;
        }
        if(a>arry[nindex])
            from = nindex+1;
        else
            to = nindex-1;
    }
    if(isFind)
        return nindex;
    return -1;
}

int main(int argc, const char * argv[]) {
    int arry[] = {1,2,3,4,5,6,7,8,9};
    int a = 5;
    int i = find(arry, sizeof(arry)/sizeof(int), a);
    cout<<a<<"在有序数组中的位置是："<<i<<endl;
    return 0;
}
```

绝对众数

算法代码：

```c

//
//  main.cpp
//  45_

//  Created by 赵琳 on 17/2/8.
//  Copyright © 2017年 赵琳. All rights reserved.

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//绝对众数
int Mode(int *a, int size) {
    int m = a[0];
    int c = 0;
    for(int i=0; i<size; i++) {
        if(c==0) {
            m = a[i];
            c++;
        } else {
            if(m == a[i])
                c++;
            else
                c--;
        }
    }
    return m;
}

int main(int argc, const char * argv[]) {
    int a[] = {8,8,1,1,1,8,1,1,6,1,8};
    int m = Mode(a, sizeof(a)/sizeof(int));
    cout<<"绝对众数是："<<m<<endl;
    return 0;
}
```

局部最大值（采用高原数组，若高原数组长度为1，即为局部最大值）

算法代码：

```c

//
//  main.cpp
//  46_
//
//  Created by 赵琳 on 17/2/8.
//  Copyright © 2017年 赵琳. All rights reserved.
//

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//局部最大值（采用高原数组，若高原数组长度为1，即为局部最大值）
int localMax(int *a, int size) {
    int left = 0, right = size-1, mid = 0;
    while (left<right) {
        mid = (left+right)/2;
        if(a[mid]>a[mid+1])
            right = mid;
        else
            left = mid+1;
    }
    return a[left];
}

int main(int argc, const char * argv[]) {
    int a[] = {1,3,5,2,4,6,7,8,2,3,5};
    int locmax = localMax(a, sizeof(a)/sizeof(int));
    cout<<"局部最大值之一是"<<locmax<<endl;
    return 0;
}
```

第一个缺失的整数

算法代码：

```c

//
//  main.cpp
//  47_
//
//  Created by 赵琳 on 17/2/8.
//  Copyright © 2017年 赵琳. All rights reserved.
//

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//第一个缺失的整数
void exchange(int &a, int &b) {
    if(a==b)    return;
    a = a^b;
    b = a^b;
    a = a^b;
}

int firstMissNumber(int *a, int size) {
    a--;
    int i = 1;
    while(i<=size) {
        if(a[i] == i)
            i++;
        else if(a[i] < i || a[i] > size || a[i] == a[a[i]]) {   //注意a[i] == a[a[i]]这个约束条件，容易忽略，对应的位置处a[i]已经有正确的值，如果即i处与a[i]对应的值一致，如果交换会造成死循环（易错点）
            a[i] = a[size];
            size--;
        } else
            exchange(a[i], a[a[i]]);
    }
    return size+1;
}

int main(int argc, const char * argv[]) {
    int a[] = {1,2,5,-1,5,7,3,8,9};
    int fm = firstMissNumber(a, sizeof(a)/sizeof(int));
    cout<<"第一个缺失的整数是："<<fm<<endl;
    return 0;
}
```

寻找循环数组的最小值：查找(由小到大)有序数组以一个点为支点进行循转后的最小值如12345-－》45123

算法代码：

```c

//
//  main.cpp
//  48_
//
//  Created by 赵琳 on 17/2/8.
//  Copyright © 2017年 赵琳. All rights reserved.
//

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//寻找循环数组的最小值：查找(由小到大)有序数组以一个点为支点进行循转后的最小值如12345-－》45123
int findMin(int *a, int size ) {
    int left = 0, right = size-1, mid = 0;
    while(left<right) {
        mid = (left+right)/2;
        if(a[mid]>a[right]) //注意mid处的值与high的值进行比较
            left = mid+1;
        else
            right = mid;
    }
    return a[left];
}

int main(int argc, const char * argv[]) {
    int a[] = {6,7,8,9,1,2,3,4,5};
    int min = findMin(a, sizeof(a)/sizeof(int));
    cout<<"循环数组的最小值："<<min<<endl;
    return 0;
}
```

返回连续数组的部分和中，绝对值最接近零的值（注意是和值，不是那一段部分数组）

算法代码：

```c

//
//  main.cpp
//  49_
//
//  Created by 赵琳 on 17/2/8.
//  Copyright © 2017年 赵琳. All rights reserved.
//

#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//返回连续数组的部分和中，绝对值最接近零的值（注意是和值，不是那一段部分数组）
int getMinPartSum(int *a, int size) {
    int *sum = new int[size+1];
    sum[0] = 0;
    for(int i=0; i<size; i++)
        sum[i+1] = sum[i]+a[i];
    
    sort(sum, sum+size+1);  //将所有可能的和进行排序后比较相邻的差
    
    int d = abs(sum[1]-sum[0]), ref = 0;
    for(int i=0; i<size; i++) {
        ref = abs(sum[i+1]-sum[i]);
        d = d>ref?ref:d;
    }
    return d;
}

int main(int argc, const char * argv[]) {
    int a[] = {1,-2,3,10,-4,7,2,-5};
    int m = getMinPartSum(a, sizeof(a)/sizeof(int));
    cout<<"连续数组部分和最接近0的值是："<<m<<endl;
    return 0;
}
```

数列部分和最大值

算法代码：

```c

//
//  main.cpp
//  50_
//
//  Created by 赵琳 on 17/2/8.
//  Copyright © 2017年 赵琳. All rights reserved.
//

#include <iostream>
#include "stdlib.h"
#include "stdio.h"

using namespace std;

//数列部分和最大值
int getMax(int *a, int size) {
    if(size<1) return 0;
    int sum = a[0]; //当前部分和
    int max = sum;  //最优解
    int from = 0, to = 0, fromtemp = 0;;
    for(int i=1; i<size; i++) {
        if(sum>0)
            sum += a[i];
        else {
            sum = a[i];
            fromtemp = i;
        }
        
        if(sum>max) {
            max = sum;
            from = fromtemp;    //from与fromtemp保持一致，fromtemp始终是当前的from
            to = i;
        }
    }
    
    cout<<"最大部分和的部分是：";
    for(int i=from; i<=to; i++) {
        cout<<a[i]<<" ";
    }
    cout<<endl;
    return max;
}

int main(int argc, const char * argv[]) {
    int a[] = {1,-2,3,10,-4,7,2,-5};
    int m = getMax(a, sizeof(a)/sizeof(int));
    cout<<"最大部分和是："<<m<<endl;
    return 0;
}
```
