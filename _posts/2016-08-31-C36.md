1.已知满二叉树的先序序列，求其后序序列（递归） 

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

//已知满二叉树的先序序列，求其后序序列（递归） 
string perToPost(string a) {
	if(a.length() <2 )    return a;
	string m = perToPost(a.substr(1,a.length()/2));
	string n = perToPost(a.substr(a.length()/2+1,a.length()/2));
	return m+n+a.substr(0,1);
}

int main(int argc, char** argv) {
	string a = "ABCDEFG";
	a = perToPost(a);
	cout<<a<<endl;
	return 0;
}
```

2. 将二叉树从左到右的叶子结点连成一个单链表

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <vector>

using namespace std;

typedef struct BiTNode {
	int data;
	struct BiTNode *lchild, *rchild;
}BiTNode, *BiTree;

BiTree T;

void createBiTree(BiTree &T) {
	T = (BiTree)malloc(sizeof(BiTNode));
	int a;
	scanf("%d", &a);
	if(a!=-1) {
		T->data = a;
		createBiTree(T->lchild);
		createBiTree(T->rchild);
	} else
		T = NULL;
}

void visit(BiTNode *p) {
	cout<<p->data<<" ";
} 

//先序遍历 
void perOrder(BiTree T) {
	if(T == NULL)    return;
	visit(T);
	perOrder(T->lchild);
	perOrder(T->rchild);
}

//中序遍历 
void inOrder(BiTree T) {
	if(T == NULL)    return;
	inOrder(T->lchild);
	visit(T);
	inOrder(T->rchild);
}

//p结点是否为叶结点 
bool isNode(BiTNode *p) {
	if(p->lchild == NULL && p->rchild == NULL) 
		return 1;
	return 0;
}

//叶结点遍历生成一个链表 ，将叶结点连成链后，原二叉树链表就乱了  
void perOrderNode(BiTree T, BiTree &p) {
	if(T == NULL)    return;
	if(isNode(T)) {
		p->rchild = T;
		p = p->rchild;
	}
	perOrderNode(T->lchild, p);
	perOrderNode(T->rchild, p);
	p->rchild = NULL;    //链表末尾置为NULL
}

//打印以p为头结点的链表 
void printfNode(BiTree p) {
	while(p->rchild!=NULL) {
		cout<<p->rchild->data<<" ";
		p = p->rchild; 
	}
}

int main(int argc, char** argv) {
	createBiTree(T);
	cout<<"先序遍历：";
	perOrder(T);
	cout<<endl<<"中序遍历：";
	inOrder(T);
	cout<<endl<<"叶结点遍历：";
	BiTree p = (BiTree)malloc(sizeof(BiTNode));    //生成链表的头结点
	BiTree p1 = p; //头指针被引用，会修改地址，头指针备份 
	perOrderNode(T, p1);
	printfNode(p);
	return 0;
}

```
