巧妙应用这种方法，可以解决大整数问题。

特别注意：当两个链表都遍历结束时，不要忘记进位还有的情况。

算法代码：

```c

//
//  main.cpp
//  16
//
//  Created by 赵琳 on 16/9/30.
//  Copyright © 2016年 赵琳. All rights reserved.
//

#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

typedef struct LinkNode {
    int data;
    struct LinkNode *next;
}LinkNode, *LinkList;

void ceratLinkList(LinkList &s) {
    int data;
    scanf("%d", &data);
    LinkList p, sTail = s;
    while(data!=-1) {
        p = (LinkList)malloc(sizeof(LinkNode));
        p->data = data;
        sTail->next = p;
        sTail = p;
        scanf("%d", &data);
    }
    sTail->next = NULL;
}

LinkList sumList(LinkList a, LinkList b) {
    LinkList sum = (LinkList)malloc(sizeof(LinkNode));
    LinkList pTail = sum;
    LinkList p1 = a->next, p2 = b->next, p;
    int c = 0, total;    //进位,和
    while(p1 && p2) {
        p = (LinkNode*)malloc(sizeof(LinkNode));
        total = p1->data + p2->data + c;
        c = total/10;
        total%=10;
        p->data = total;
        pTail->next = p;
        pTail = p;
        p1 = p1->next;
        p2 = p2->next;
    }
    p1 = p1!=NULL ?p1:p2;
    while (p1) {    //某一个list遍历完，另一个还没有遍历完的情况
        p = (LinkNode*)malloc(sizeof(LinkNode));
        total = p1->data + c;
        c = total/10;
        total%=10;
        p->data = total;
        pTail->next = p;
        pTail = p;
        p1 = p1->next;
    }
    if(c!=0) {    //最后仍有一位进位
        p = (LinkNode*)malloc(sizeof(LinkNode));
        p->data = c;
        pTail->next = p;
        pTail = p;
    }
    pTail->next = NULL;
    return sum;
}

void coutLinkList(LinkList s) {
    while(s->next!=NULL) {
        cout<<s->next->data<<" ";
        s = s->next;
    }
    cout<<endl;
}

int main(int argc, const char * argv[]) {
    LinkList a = (LinkList)malloc(sizeof(LinkNode));
    LinkList b = (LinkList)malloc(sizeof(LinkNode));
    ceratLinkList(a);
    ceratLinkList(b);
    LinkList sum = sumList(a, b);
    coutLinkList(sum);
    return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>

using namespace std;

typedef struct LinkNode {
    int data;
    struct LinkNode *next;
}LinkNode, *LinkList;

void ceratLinkList(LinkList &s) {
    int data;
    scanf("%d", &data);
    LinkList p, sTail = s;
    while(data!=-1) {
        p = (LinkList)malloc(sizeof(LinkNode));
        p->data = data;
        sTail->next = p;
        sTail = p;
        scanf("%d", &data);
    }
    sTail->next = NULL;
}

//翻转链表第m个元素到第n个元素
void reverseLinkList(LinkList &s, int m, int n) {
    if(m >= n) return;
    LinkList p = s, pHead, pTail, temp;
    int count = 1;
    while(p->next!=NULL && count < m) {
        count++;
        p = p->next;
    }   //p后面的元素开始翻转
    pHead = p->next;
    if(pHead!=NULL)
        temp = pHead->next;
    else
        return;
    while (temp && count < n) {
        count++;
        pTail = temp->next;
        temp->next = p->next;
        p->next = temp;
        temp = pTail;
    }
    
    if(temp)
        pHead->next = temp;
    else
        pHead->next = NULL;
}

void coutLinkList(LinkList s) {
    while(s->next!=NULL) {
        cout<<s->next->data<<" ";
        s = s->next;
    }
    cout<<endl;
}

int main(int argc, const char * argv[]) {
    LinkList a = (LinkList)malloc(sizeof(LinkNode));
    ceratLinkList(a);
    reverseLinkList(a, 2, 5);
    coutLinkList(a);
    return 0;
}
```

```c
//已排序链表去重
void reverseLinkList(LinkList &s) {
    if(s->next == NULL) return;
    LinkList p = s->next, temp;
    while(p->next!=NULL) {
        if(p->data!=p->next->data)
            p = p->next;
        else {  //  记得把删除的元素释放空间
            temp = p->next;
            p->next = p->next->next;
            free(temp);
        }
    }
}
```

```c
//已排序链表去重(将重复元素全部删除)
void reverseLinkList2(LinkList &s) {
    if(s->next == NULL) return;
    LinkList p = s, temp;
    while(p->next && p->next->next) {
        if(p->next->data!=p->next->next->data)
            p = p->next;
        else {  //  记得把删除的元素释放空间
            while(p->next->next && (p->next->data == p->next->next->data)) {
                temp = p->next->next;
                p->next->next = p->next->next->next;
                free(temp);
            }
            temp = p->next;
            p->next = p->next->next;
            free(temp);
        }
    }
}

void coutLinkList(LinkList s) {
    while(s->next!=NULL) {
        cout<<s->next->data<<" ";
        s = s->next;
    }
    cout<<endl;
```
