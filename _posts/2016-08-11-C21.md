使用工具：DEV C++

题目描述：  

如何利用栈的运算实现队列的运算(当S2不为空的时候，入队和出队互不干涉)

```c

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#define MaxSize 5 
/*
如何利用栈的运算实现队列的运算 
*/
using namespace std;

//栈的结构体 
typedef struct {
	int data[MaxSize];
	int top; 
}Stack;

Stack S1;
Stack S2;

//初始化栈 
void InitStack() {
	S1.top = 0; 
	S2.top = 0;
}

//判栈空
bool StackEmpty(Stack &S) {
	if(S.top == 0)    return true;
	return false;
}

//判栈满 
bool StackOverflow(Stack &S) {
	if(S.top == MaxSize)    return true;
	return false;
}

//进栈
bool Push(Stack &S, int x) {
	if(StackOverflow(S))    return false;
	S.data[S.top++] = x;
	return true;	
}

//出栈 
bool Pop(Stack &S, int &x) {
	if(StackEmpty(S))   return false;
	x = S.data[--S.top];
	return true;	
}

//模拟判队空
bool QueueEmpty() {
	if(StackEmpty(S1) && StackEmpty(S2))    return true;
	return false;
}

//模拟队列进队
bool EnQueue(int x) {
	return Push(S1, x);
}

//模拟队列出队 
bool DeQueue(int &x) {
	x = -1;
	if(QueueEmpty())    return false;
	if(!StackEmpty(S2)) {
		Pop(S2, x);
	} else {//S2为空，将S1出栈，进入S2 	
		while(!StackEmpty(S1)) {
			Pop(S1, x);
			Push(S2, x);
		}
		Pop(S2, x);
	}
	return true;
} 

int main(int argc, char** argv) {
	bool flag;
	int x;
	InitStack();
	flag = EnQueue(1);
	cout<<"1 入队状态："<<flag<<endl;
	flag = EnQueue(2);
	cout<<"2 入队状态："<<flag<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = EnQueue(3);
	cout<<"3 入队状态："<<flag<<endl;
	flag = EnQueue(4);
	cout<<"4 入队状态："<<flag<<endl;
	flag = EnQueue(5);
	cout<<"5 入队状态："<<flag<<endl;
	flag = EnQueue(6);
	cout<<"6 入队状态："<<flag<<endl;
	flag = EnQueue(7);
	cout<<"7 入队状态："<<flag<<endl;
	flag = EnQueue(8);
	cout<<"8 入队状态："<<flag<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	flag = DeQueue(x);
	cout<<"出队状态："<<flag<<"；出队值："<<x<<endl;
	return 0;
}
```
