```c
#include <iostream>
#include "stdio.h"
#include "stdlib.h"

using namespace std;

//最长回文子串(Manacher)
void Manacher(char *s, int size) {
    int *p = new int[size];
    p[0] = 1;
    int id = 0;
    int mx = 1;
    for(int i=1; i<size; i++) {
        if(mx>i)
            p[i] = (mx-i)>p[2*id-i]?p[2*id-i]:(mx-i);
        else
            p[i] = 1;
        for(; s[i+p[i]]==s[i-p[i]]; p[i]++);
        id = mx>=(i+p[i])?id:i;
        mx = mx>=(i+p[i])?mx:(i+p[i]);
    }
    int max = 0;
    for(int i=0; i<size; i++)
        max = p[max]>p[i]?max:i;
    for(int i=max-p[max]+1; i<max+p[max]; i++)
        if(s[i]!=s[0])
            cout<<s[i];
    cout<<endl;
}

int main(int argc, const char * argv[]) {
    char s[] = "#1#2#2#1#2#3#2#1#";
    Manacher(s, (int)strlen(s));
    return 0;
}
```

```c
//三字母字符串组合（相连三个字符不能完全相同，长度为n，求个数）

int getNum(int n) {
    if(n<1) return 0;
    int a = 3;
    int b = 0;
    int t;
    for(int i=2; i<=n; i++) {
        t = a;
        a = 2*a + 2*b;
        b = t;
    }
    return a+b;
}

int main(int argc, const char * argv[]) {
    int n = 4;
    int a = getNum(n);
    cout<<"n为"<<n<<"时，满足条件字符串的个数："<<a<<endl;
    return 0;
}
```

```c

//利用矩阵化为复杂度为logn的算法
typedef struct Matrix22 {
    //第一行
    int a;
    int b;
    //第二行
    int c;
    int d;
    Matrix22(int x, int y, int z, int o):a(x),b(y),c(z),d(o){}
    void setMatrix22(int x, int y, int z, int o) {
        a = x;
        b = y;
        c = z;
        d = o;
    }
}Matrix22;

void MatrixMulti(Matrix22 &m, Matrix22 n) {
    int a = m.a * n.a + m.b * n.c;
    int b = m.a * n.b + m.b * n.d;
    int c = m.c * n.a + m.d * n.c;
    int d = m.c * n.b + m.d * n.d;
    m.setMatrix22(a, b, c, d);
}

void MatrixN(Matrix22 &m, int n) {
    if(n==0)
        m.setMatrix22(1,0,0,1);
    else if(n!=1) {
        if(n%2!=0) {    //n为奇数
            Matrix22 x = m;
            MatrixN(m, n/2);
            MatrixMulti(m, m);
            MatrixMulti(m, x);
        } else {    //n为偶数
            MatrixN(m, n/2);
            MatrixMulti(m, m);
        }
    }
}

int getNum(int n) {
    int d0 = 3;
    int d1 = 0;
    Matrix22 m(2,2,1,0);
    MatrixN(m, n-1);
    return d0*m.a + d1*m.b + d0*m.c + d1*m.d;
}
```
